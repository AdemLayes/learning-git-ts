- git merge 'branch name'

- git commit -m "message"
  git commit
  git commit -a
- git diff
  git diff --staged
- git rm --cached 'file name'

- git push -u 'remote branch name' 'branch name / master name'
- git pull 'remote branch name' 'branch name / master name'
