Git commends i learned so far:

- git config --global user.name 'name'
  git config --global user.email 'email'

- git mkdir 'derectory name'
- git cd 'derectory name'
- git touch 'file name'
- git init
  git init 'repo name'

- git add 'file name'
  git add .
  git add *.html / *index
- git commit -m "message"
  git commit
  git commit -a

- git branch
  git branch 'branch name'
  git branch -d 'branch name'
- git checkout 'branch name / master / ID identifier'
  git checkout -b 'branch name'
- git merge 'branch name'

- git status
  git status -s
- git log
  git log --oneline
  git log --reverse
- git show HEAD~'number'
  git show 'ID identifier'
- git diff
  git diff --staged
- ls
  ls -a
  git ls-files
  git ls-tree 'master / HEAD~'number' / ID identifier'

- git rm --cached 'file name'
  git rm 'file name'
- git restore --staged 'file name'
  git restore 'file name'
- git clean -fd

- git remote -v
  git remote add origin 'url(from gitHub)'

- git push add --set-upsteam origin master
- git push -u origin master
- git pull 'remote branch name' 'branch name / master name'
